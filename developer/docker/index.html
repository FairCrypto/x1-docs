# Run with Docker

See the Docker packages hereopen in new window.

Pull the latest image

docker pull ghcr.io/faircrypto/go-x1:latest

Create the data directory

mkdir -p $HOME/.x1

Run the container

Full Node
docker run -d --name x1 \
  -p 5050:5050 \
  -v $HOME/.x1:/root/.x1 \
  ghcr.io/faircrypto/go-x1:latest \
      --testnet \
      --syncmode snap
API Node
docker run -d --name x1 \
  -p 5050:5050 \
  -p 8545:8545 \
  -p 8546:8546 \
  -v $HOME/.x1:/root/.x1 \
  ghcr.io/faircrypto/go-x1:latest \
      --testnet \
      --syncmode snap \
      --http \
      --http.port 8545 \
      --http.addr 0.0.0.0 \
      --http.vhosts "*" \
      --http.corsdomain "*" \
      --ws \
      --ws.addr 0.0.0.0 \
      --ws.port 8546 \
      --ws.origins "*"
Archive Node
mkdir -p $HOME/.x1

# Run the container with the data directory mounted
docker run -d --name x1 \
  -p 5050:5050 \
  -v $HOME/.x1:/root/.x1 \
  ghcr.io/faircrypto/go-x1:latest \
      --testnet \
      --syncmode full \
      --gcmode archive
